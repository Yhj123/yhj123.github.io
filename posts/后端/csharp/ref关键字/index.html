#include <stdio.h>
#include <winsock2.h>
#pragma comment(lib, "ws2_32")

void bigOrSmall()
{
 int byte = 0x12345678;
 byte = htonl(byte);
 /*
 1位十六进制数可以用4位二进制数来表示
 16进制1位 = 2进制4位
 */
 char * ptr = (char *)&byte;
 printf("ptr  :%lu, byte  :%x \n",ptr,*ptr);
 printf("ptr+1:%lu, byte+1:%x, \n",(ptr+1),*(ptr+1));
}
int main() {
    bigOrSmall();
}
